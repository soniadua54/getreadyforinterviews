from pydantic import BaseModel, validator

app = FastAPI(title="Hotel API", version="1.0.0")

class CreateBookingRequest(BaseModel):
    room_id: str
    check_in: str
    check_out: str
    guest_count: int

    @validator('guest_count')
    def validate_guests(cls, v):
        if v < 1 or v > 10:
            raise ValueError('guest_count must be between 1 and 10')
        return v

# Async route — non-blocking I/O
@app.post("/api/v1/bookings", status_code=201)

Python Concurrency — Threading vs Asyncio vs Multiprocessing
The GIL, async event loop, and when to use each model
Advanced▼
# The GIL (Global Interpreter Lock) — the most-asked Python internals question
# CPython only allows ONE thread to execute Python bytecode at a time
# This means threading doesn't give true parallelism for CPU-bound work

# Three concurrency models in Python:

# 1. threading — I/O-bound tasks, GIL released during I/O
import threading

def fetch_tenant_data(tenant_id: str):
    response = requests.get(f"/api/tenants/{tenant_id}")  # GIL released here
    return response.json()

# Multiple threads — OK for HTTP calls, file I/O
threads = [threading.Thread(target=fetch_tenant_data, args=(tid,)) for tid in tenant_ids]
for t in threads: t.start()
for t in threads: t.join()

# 2. asyncio — I/O-bound, single thread, cooperative scheduling
# Better than threading for high concurrency (1000s of connections)
import asyncio
import aiohttp

async def fetch_all(tenant_ids: list[str]):
    async with aiohttp.ClientSession() as session:
        tasks = [fetch_one(session, tid) for tid in tenant_ids]
        results = await asyncio.gather(*tasks)  # run all concurrently
    return results

async def fetch_one(session: aiohttp.ClientSession, tenant_id: str):
    async with session.get(f"/api/tenants/{tenant_id}") as resp:
        return await resp.json()

# 3. multiprocessing — CPU-bound tasks, bypasses GIL
from multiprocessing import Pool

def compute_forecast(data: list) -> float:
    # Heavy ML computation — benefits from true parallelism
    return run_lightgbm_model(data)

with Pool(processes=4) as pool:  # 4 CPU cores
    results = pool.map(compute_forecast, datasets)  # parallel execution

# When to use which:
# Threading:       I/O-bound + sharing state between threads (use locks)
# Asyncio:         I/O-bound + high concurrency + you control the codebase
# Multiprocessing: CPU-bound (data processing, model inference)
#
# Rule of thumb: "If you're waiting (network/disk) → asyncio. If you're computing → multiprocessing."
At BRPL (your forecasting work): LightGBM training is CPU-bound → multiprocessing. Weather API fetching is I/O-bound → asyncio with aiohttp would be correct. The GIL is why Python's threading is fine for web servers (waiting on DB) but useless for parallel number crunching.

# Generators — lazy iteration (don't load everything into memory)
# Use yield instead of return — returns values one at a time

def read_large_csv(filepath: str):
    """Generator — yields one row at a time, never loads whole file into memory"""
    with open(filepath) as f:
        header = next(f)
        for line in f:
            yield line.strip().split(',')  # yield = pauses and returns value

# Process 10M row CSV without running out of RAM
for row in read_large_csv('load_data_2024.csv'):
    process_row(row)  # only ONE row in memory at a time

# Generator expression (like list comprehension but lazy)
squares = (x**2 for x in range(1_000_000))  # no memory allocation yet
total = sum(squares)  # computed on demand

# Decorators — wrap functions with reusable behaviour

import functools, time, logging

def retry(max_attempts: int = 3, delay: float = 1.0):
    """Decorator factory — retries function on exception"""
    def decorator(func):
        @functools.wraps(func)  # preserve function name and docstring
        def wrapper(*args, **kwargs):
            for attempt in range(max_attempts):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    if attempt == max_attempts - 1:
                        raise  # re-raise on final attempt
                    time.sleep(delay)
        return wrapper
    return decorator

@retry(max_attempts=3, delay=2.0)
def fetch_weather(station_id: str) -> dict:
    return requests.get(f"/weather/{station_id}").json()
