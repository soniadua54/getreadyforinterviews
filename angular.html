// Angular component — the building block of every UI
import { Component, OnInit, OnDestroy } from '@angular/core';
import { ActivatedRoute } from '@angular/router';

@Component({
  selector: 'app-hotel-details',         // <app-hotel-details> in templates
  templateUrl: './hotel-details.component.html',
  styleUrls: ['./hotel-details.component.scss'],
  changeDetection: ChangeDetectionStrategy.OnPush  // perf optimization
})
export class HotelDetailsComponent implements OnInit, OnDestroy {
  hotel: Hotel | null = null;
  loading = false;
  private destroy$ = new Subject<void>();

  // Dependency injection via constructor
  constructor(
    private hotelService: HotelService,
    private route: ActivatedRoute,
    private router: Router
  ) {}

  ngOnInit(): void {
    // takeUntil(destroy$) — auto-unsubscribe on component destroy
    this.route.params.pipe(
      switchMap(params => {
        this.loading = true;
        return this.hotelService.getById(params['id']);
      }),
      takeUntil(this.destroy$)
    ).subscribe({
      next: hotel => { this.hotel = hotel; this.loading = false; },
      error: err => { console.error(err); this.loading = false; }
    });
  }

  ngOnDestroy(): void {
    this.destroy$.next();    // emit signal
    this.destroy$.complete(); // complete the subject
    // All subscriptions with takeUntil(destroy$) auto-unsubscribe
  }
}
// Service — shared business logic and HTTP calls
@Injectable({ providedIn: 'root' })  // singleton, available everywhere
export class HotelService {
  private apiUrl = '/api/v1/hotels';

  constructor(private http: HttpClient) {}

  getById(id: string): Observable<Hotel> {
    return this.http.get<Hotel>(`${this.apiUrl}/${id}`).pipe(
      retry({ count: 2, delay: 1000 }),         // retry twice on error
      catchError(err => {
        if (err.status === 404) return throwError(() => new Error('Not found'));
        return throwError(() => err);
      })
    );
  }

  search(params: SearchParams): Observable<Hotel[]> {
    return this.http.get<Hotel[]>(this.apiUrl, { params: { ...params } });
  }
}

RxJS — Observables, Operators, Common Patterns
The most important Angular concept for backend engineers to understand
Intermediate▼
// RxJS key operators — you WILL be asked about these

// map — transform each value
this.hotelService.getAll().pipe(
  map(hotels => hotels.filter(h => h.starRating >= 4))
);

// switchMap — cancel previous HTTP call on new search input
// Perfect for search-as-you-type (autocomplete)
this.searchControl.valueChanges.pipe(
  debounceTime(300),           // wait 300ms after user stops typing
  distinctUntilChanged(),        // don't search if same value
  switchMap(query =>             // cancel prev request, start new one
    this.hotelService.search(query)
  ),
  takeUntil(this.destroy$)
).subscribe(results => this.hotels = results);

// mergeMap — run all inner observables concurrently
// Use when order doesn't matter, all should run in parallel
hotelIds$.pipe(
  mergeMap(id => this.hotelService.getById(id))
);

// concatMap — run inner observables in ORDER
// Use when operations must be sequential
bookingIds$.pipe(
  concatMap(id => this.bookingService.confirm(id))
);

// forkJoin — like Promise.all — wait for ALL to complete
forkJoin({
  hotel: this.hotelService.getById(id),
  rooms: this.roomService.getByHotel(id),
  reviews: this.reviewService.getByHotel(id)
}).subscribe(({ hotel, rooms, reviews }) => {
  this.hotelData = { hotel, rooms, reviews };
});

// Subject types
// Subject       — multicast, no initial value, no replay
// BehaviorSubject — has initial value, new subscribers get LAST emitted value
// ReplaySubject — replays last N values to new subscribers
// AsyncSubject  — emits only last value when completed

// BehaviorSubject — most common for state
private userState = new BehaviorSubject<User | null>(null);
currentUser$ = this.userState.asObservable();

Routing — Lazy Loading & Route Guards
Reduce initial bundle size with lazy-loaded feature modules
Intermediate▼
// app-routing.module.ts — lazy loading feature modules
const routes: Routes = [
  {
    path: 'hotels',
    // loadChildren: loaded ONLY when user navigates to /hotels
    // This splits HotelModule into a separate JS chunk
    // Result: smaller initial bundle, faster app startup
    loadChildren: () =>
      import('./hotels/hotel.module').then(m => m.HotelModule)
  },
  {
    path: 'admin',
    loadChildren: () =>
      import('./admin/admin.module').then(m => m.AdminModule),
    canActivate: [AuthGuard]       // protect admin routes
  },
  { path: '', redirectTo: '/hotels', pathMatch: 'full' },
  { path: '**', component: NotFoundComponent }  // 404 fallback
];

// Route Guard — prevent unauthorized access
@Injectable({ providedIn: 'root' })
export class AuthGuard implements CanActivate {
  constructor(private auth: AuthService, private router: Router) {}

  canActivate(route: ActivatedRouteSnapshot, state: RouterStateSnapshot): boolean {
    if (this.auth.isAuthenticated()) {
      return true;
    }
    this.router.navigate(['/login'], { queryParams: { returnUrl: state.url } });
    return false;
  }
}
Why lazy loading? Without it, Angular bundles your entire app into one JS file. With lazy loading, each feature module is a separate chunk. The initial load only downloads the home page code; admin code is fetched only when someone navigates to /admin. This matters at Agoda scale — every KB saved = faster load for users on mobile networks in SEA.

HTTP Interceptors — Auth Token, Error Handling, Loading
Middleware for all HTTP requests — add auth token, handle 401, show loading state
Intermediate▼
// AuthInterceptor — adds JWT to every outbound request
@Injectable()
export class AuthInterceptor implements HttpInterceptor {
  constructor(private auth: AuthService, private router: Router) {}

  intercept(req: HttpRequest<any>, next: HttpHandler): Observable<HttpEvent<any>> {
    const token = this.auth.getToken();

    // Clone the request (HttpRequest is immutable)
    const authReq = token
      ? req.clone({ headers: req.headers.set('Authorization', `Bearer ${token}`) })
      : req;

    return next.handle(authReq).pipe(
      catchError((err: HttpErrorResponse) => {
        if (err.status === 401) {
          // Token expired — redirect to login
          this.auth.logout();
          this.router.navigate(['/login']);
        }
        if (err.status === 403) {
          this.router.navigate(['/unauthorized']);
        }
        return throwError(() => err);
      })
    );
  }
}

// Register in AppModule providers:
providers: [
  { provide: HTTP_INTERCEPTORS, useClass: AuthInterceptor, multi: true }
]
// multi: true — allows multiple interceptors (they form a chain)
💡 Interview answer
HTTP interceptors are Angular's equivalent of middleware. At Biz2Credit (HSBC lending), we used interceptors for: (1) adding the Auth0 JWT to every API call, (2) handling global 401 redirects to re-login, and (3) showing/hiding a global loading spinner. Without interceptors, you'd have to add auth headers in every single service method — interceptors make it DRY.

Reactive Forms
// Default change detection: Angular checks ALL components on any event/async op
// OnPush: only checks component when:
//   1. @Input() reference changes (not mutation)
//   2. An Observable emits (with async pipe)
//   3. You manually call markForCheck()

@Component({
  changeDetection: ChangeDetectionStrategy.OnPush  // ← performance optimization
})
export class HotelListComponent {
  // Use async pipe — automatically handles subscription AND OnPush detection
  hotels$ = this.hotelService.getAll();  // Observable
}

// Template: async pipe + OnPush = no manual subscriptions
// <ng-container *ngIf="hotels$ | async as hotels">
//   <app-hotel-card *ngFor="let h of hotels" [hotel]="h"></app-hotel-card>
// </ng-container>

// Key rule: with OnPush, mutating an object WON'T trigger re-render
// WRONG (mutation — OnPush won't detect):
this.hotel.price = 200;            // same object reference, no change detected

// CORRECT (new reference — OnPush detects input change):
this.hotel = { ...this.hotel, price: 200 };  // spread = new object = new reference
// Reactive Forms — programmatic control vs template-driven (simpler but less powerful)
import { FormBuilder, Validators, AbstractControl } from '@angular/forms';

@Component({ ... })
export class BookingFormComponent {
  bookingForm = this.fb.group({
    guestName:  ['', [Validators.required, Validators.minLength(2)]],
    email:      ['', [Validators.required, Validators.email]],
    checkIn:    [null, Validators.required],
    checkOut:   [null, Validators.required],
    guestCount: [1,    [Validators.min(1), Validators.max(10)]]
  }, { validators: this.dateRangeValidator });  // cross-field validator

  constructor(private fb: FormBuilder) {}

  // Custom cross-field validator (check-out must be after check-in)
  private dateRangeValidator(form: AbstractControl) {
    const checkIn = form.get('checkIn')?.value;
    const checkOut = form.get('checkOut')?.value;
    if (checkIn && checkOut && checkOut <= checkIn) {
      return { invalidDateRange: true };  // error object
    }
    return null;                              // null = valid
  }

  onSubmit() {
    if (this.bookingForm.valid) {
      // bookingForm.value is typed — no casting needed
      this.bookingService.create(this.bookingForm.value).subscribe();
    } else {
      this.bookingForm.markAllAsTouched();  // show all validation errors
    }
  }
}
