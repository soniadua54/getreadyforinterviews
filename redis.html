# String — most common type
SET hotel:123 '{"name":"Grand Hotel","city":"Bangkok"}' EX 3600  # set with 1hr TTL
GET hotel:123
SETNX lock:room:456:2026-07-01 "userId-789"  # set ONLY if not exists (lock)
INCR booking:counter:today         # atomic increment
INCRBY booking:counter:today 5

# Hash — efficient for objects with multiple fields
HSET user:123 name "Sonia" email "sonia@example.com" role "ADMIN"
HGET user:123 name              # get single field
HMGET user:123 name email       # get multiple fields
HGETALL user:123                # get all fields
HINCRBY user:123 loginCount 1  # increment a field

# Sorted Set — perfect for sliding window rate limiting
# Score = timestamp, member = request UUID
ZADD user:123:requests 1719910800000 "req-uuid-1"   # score = timestamp ms
ZREMRANGEBYSCORE user:123:requests 0 1719910740000  # remove older than 1 min
ZCARD user:123:requests                              # count in window

# List as queue
LPUSH job:queue "job-1" "job-2"   # push to front
RPOP job:queue                    # pop from back (FIFO)
BRPOP job:queue 30                # blocking pop — waits up to 30s

# Set for online users
SADD online:users "user-123"
SREM online:users "user-123"       # remove (went offline)
SISMEMBER online:users "user-456"  # check membership
SCARD online:users                 # count online users
# TTL — time to live
SET session:token123 "user-data" EX 1800        # expire in 30 minutes
SET session:token123 "user-data" PX 1800000     # expire in 1800000 milliseconds
EXPIRE session:token123 1800                     # set TTL on existing key
PERSIST session:token123                         # remove TTL (make permanent)
TTL session:token123                             # check remaining TTL in seconds
PTTL session:token123                            # check remaining TTL in milliseconds

# Eviction policies — what happens when memory is full
# Set in redis.conf or via CONFIG SET
# noeviction     → return error when memory full (default for non-cache)
# allkeys-lru    → evict least recently used key from ALL keys
# volatile-lru   → evict LRU from keys WITH TTL set
# allkeys-lfu    → evict least frequently used (better than LRU for skewed access)
# volatile-ttl   → evict key with shortest TTL first

# For a hotel cache:
maxmemory 2gb
maxmemory-policy allkeys-lru       # most common for cache use case

# Spring Boot config
# spring.redis.timeout=2000ms  ← fail fast if Redis slow
# spring.cache.redis.time-to-live=5m  ← default TTL

Distributed Locks — Preventing Double Booking
SETNX pattern · Redlock algorithm · Lock expiry
Intermediate▼
Why needed
Implementation
Java (Redisson)
The problem without distributed locks
500 users request the last available room simultaneously. All read available_count = 1. All pass the check. All create a booking. Result: 500 confirmed bookings for 1 room. A single-process lock (synchronized/mutex) only works within one JVM. With multiple service instances (3 pods), they don't share memory.
Redis distributed lock: Only one process can hold the lock at a time, across ALL service instances. The lock is a key in Redis. If the key exists = locked. SETNX (SET if Not eXists) is atomic — only the first caller sets it.
# SETNX-based distributed lock
# Acquire lock: SET key value NX PX milliseconds
# NX = only set if Not eXists
# PX = auto-expire to prevent deadlock if process crashes

SET lock:room:123:2026-07-01 "owner-uuid-abc" NX PX 30000
# Returns "OK" if lock acquired
# Returns nil if someone else holds lock

# Release lock — MUST only delete if we own it!
# Wrong: just DELETE the key — could delete someone else's lock
# Right: Lua script for atomic check-and-delete

# lua-release-lock.lua
if redis.call("GET", KEYS[1]) == ARGV[1] then
    return redis.call("DEL", KEYS[1])
else
    return 0  -- someone else owns it, don't delete
end

# Run Lua script
EVAL "if redis.call('GET',KEYS[1])==ARGV[1] then return redis.call('DEL',KEYS[1]) else return 0 end" \
  1 lock:room:123:2026-07-01 "owner-uuid-abc"

# The lock value MUST be a unique owner ID (UUID)
# So you can only release locks YOU own
# Prevents: process A acquires, process A's lock expires, process B acquires,
#           process A tries to release B's lock → the UUID check prevents this
// Redisson — production-grade Redis client with lock abstraction
// Implements Redlock algorithm (multi-node quorum)

@Service
public class BookingService {
    private final RedissonClient redisson;
    private final BookingRepository bookingRepo;

    public Booking createBooking(CreateBookingRequest req) {
        String lockKey = "lock:room:" + req.getRoomId() + ":" + req.getCheckIn();
        RLock lock = redisson.getLock(lockKey);

        try {
            // Try to acquire lock, wait max 5s, auto-release after 30s
            boolean acquired = lock.tryLock(5, 30, TimeUnit.SECONDS);
            if (!acquired) {
                throw new RoomUnavailableException("Room is being booked, please retry");
            }

            // Inside lock — only ONE thread across all pods runs this
            int available = inventoryService.getAvailability(req.getRoomId(), req.getCheckIn());
            if (available <= 0) throw new RoomUnavailableException("No rooms available");

            inventoryService.decrementAvailability(req.getRoomId(), req.getCheckIn());
            return bookingRepo.save(new Booking(req));

        } catch (InterruptedException e) {
            Thread.currentThread().interrupt();
            throw new BookingException("Booking interrupted");
        } finally {
            // Always release if we own it
            if (lock.isHeldByCurrentThread()) lock.unlock();
        }
    }
}

Rate Limiting with Redis
Token bucket · Sliding window — how to protect your APIs
Intermediate▼ 
// Rate limiter with Redis Sorted Set (sliding window)
@Service
public class RateLimiter {
    private final RedisTemplate<String, String> redis;
    private final int MAX_REQUESTS = 100;
    private final long WINDOW_MS = 60_000; // 1 minute

    public boolean isAllowed(String userId) {
        String key = "rate:" + userId;
        long now = System.currentTimeMillis();
        long windowStart = now - WINDOW_MS;

        return redis.execute(new SessionCallback<Boolean>() {
            public Boolean execute(RedisOperations ops) {
                ops.multi(); // atomic transaction
                ops.opsForZSet().removeRangeByScore(key, 0, windowStart); // remove old
                ops.opsForZSet().add(key, UUID.randomUUID().toString(), now);
                ops.opsForZSet().size(key);
                ops.expire(key, Duration.ofMinutes(2));
                List<Object> results = ops.exec();
                long count = (Long) results.get(2);
                return count <= MAX_REQUESTS;
            }
        });
    }
}

// In Spring Boot: Bucket4j + Redis for token bucket
@Bean
public BucketProxyManager bucketManager(RedissonClient redisson) {
    return new RedissonBasedProxyManager(redisson);
}

// In filter:
BucketConfiguration config = BucketConfiguration.builder()
    .addLimit(Bandwidth.classic(100, Refill.intervally(100, Duration.ofMinutes(1))))
    .build();

Bucket bucket = bucketManager.builder()
    .build(userId, () -> config);

if (!bucket.tryConsume(1)) {
    response.sendError(429, "Too Many Requests");
    return;
}

Pub/Sub & Redis Streams
Real-time messaging without Kafka complexity
Intermediate▼ 
# Redis Pub/Sub — fire-and-forget messaging
# Publisher
PUBLISH booking.events '{"type":"BOOKING_CREATED","bookingId":"123","userId":"456"}'

# Subscriber (in another process)
SUBSCRIBE booking.events
PSUBSCRIBE booking.*    # pattern subscribe — matches booking.events, booking.cancelled etc

# Limitation: messages lost if subscriber is offline (not durable)
# Use Redis Streams for durability

──────────────────────────────────────

# Redis Streams — durable, consumer groups (like Kafka but simpler)
# Add event to stream
XADD booking:events * type BOOKING_CREATED bookingId 123 userId 456
# * = auto-generate ID (timestamp-sequence: 1719910800000-0)

# Create consumer group (enables parallel consumers + offset tracking)
XGROUP CREATE booking:events notification-service $ MKSTREAM
# $ = start from latest message

# Consumer reads from group
XREADGROUP GROUP notification-service consumer-1 COUNT 10 BLOCK 2000 STREAMS booking:events >
# > = messages not yet delivered to this group

# Acknowledge processing
XACK booking:events notification-service 1719910800000-0

# Spring Boot Redis Streams
@Bean
public Subscription subscription(RedisConnectionFactory factory) {
    StreamMessageListenerContainer<String, MapRecord<String,String,String>> container =
        StreamMessageListenerContainer.create(factory);

    container.receive(
        Consumer.from("notification-service", "consumer-1"),
        StreamOffset.create("booking:events", ReadOffset.lastConsumed()),
        new BookingEventListener()
    );
    container.start();
    return container;
}

Redis Cluster & High Availability
Replication · Sentinel · Cluster mode · persistence
Advanced▼
Redis high availability options
Option 1: Replication (Master + Replicas) Master ← all writes Replica ← sync from master, handle reads Replica ← Limitation: if master dies, manual failover needed Option 2: Redis Sentinel (automatic failover) Master + 2 Replicas + 3 Sentinels Sentinels monitor master, vote to promote replica if master fails Automatic failover in ~30 seconds Option 3: Redis Cluster (sharding + HA) 6 nodes: 3 masters + 3 replicas (1 replica per master) Data sharded across masters using 16384 hash slots key → hash → slot → master node If a master dies: its replica promoted automatically Can scale horizontally by adding more shards
# Persistence options (important for Staff-level answers)
# RDB (snapshot) — point-in-time snapshot
save 900 1          # save if 1 key changed in 900s
save 300 10         # save if 10 keys changed in 300s
# Pros: compact file, fast restart. Cons: data loss between snapshots.

# AOF (Append Only File) — log every write operation
appendonly yes
appendfsync everysec   # fsync every second (balance durability vs performance)
# Pros: minimal data loss. Cons: larger file, slower restart.

# Production: use both RDB + AOF for maximum durability
