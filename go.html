// Go basics — the syntax you need to know

// No classes — use structs + methods
type Hotel struct {
    ID        string
    Name      string
    City      string
    StarRating int
}

// Method on struct (Go's version of a class method)
func (h Hotel) DisplayName() string {
    return fmt.Sprintf("%s (%d stars)", h.Name, h.StarRating)
}

// Interfaces — implicit implementation (no 'implements' keyword)
type Repository interface {
    FindByID(id string) (*Hotel, error)
    Save(hotel *Hotel) error
}
// Any struct with FindByID and Save methods automatically implements Repository

// Error handling — explicit, not exceptions
func GetHotel(id string) (*Hotel, error) {
    hotel, err := db.Query("SELECT * FROM hotels WHERE id = ?", id)
    if err != nil {
        return nil, fmt.Errorf("fetching hotel %s: %w", id, err)  // %w wraps error
    }
    if hotel == nil {
        return nil, ErrNotFound
    }
    return hotel, nil
}

// Custom errors
var ErrNotFound = errors.New("hotel not found")

// Sentinel error check
hotel, err := GetHotel("123")
if errors.Is(err, ErrNotFound) {
    // handle not found specifically
}

// Slices (like Java ArrayList)
hotels := []Hotel{}
hotels = append(hotels, Hotel{ID: "1", Name: "Grand"})

// Maps (like Java HashMap)
byID := map[string]Hotel{"1": {ID: "1", Name: "Grand"}}
hotel, ok := byID["1"]  // ok = false if key doesn't exist
if !ok { /* not found */ }

What is a goroutine?
A goroutine is a lightweight thread managed by the Go runtime. Unlike OS threads (~1MB stack, expensive context switch), goroutines start with a ~2KB stack that grows as needed. You can run millions of goroutines on a single machine. The Go runtime schedules them onto OS threads (M:N threading).
// Start a goroutine with 'go' keyword
func main() {
    // This runs in a NEW goroutine (concurrent)
    go sendEmail("user@example.com")
    // main() continues immediately — doesn't wait for email

    // Anonymous goroutine
    go func() {
        fmt.Println("running in background")
    }()

    // PROBLEM: main exits → all goroutines killed
    // Solution: use sync.WaitGroup or channels to wait
}

// WaitGroup — wait for multiple goroutines to finish
func processBookings(bookings []Booking) {
    var wg sync.WaitGroup

    for _, booking := range bookings {
        wg.Add(1)               // increment counter
        go func(b Booking) {
            defer wg.Done()    // decrement when goroutine exits
            processBooking(b)
        }(booking)              // pass booking as arg (avoid closure bug)
    }

    wg.Wait() // blocks until all goroutines call Done()
    fmt.Println("all bookings processed")
}
// Channel — typed pipe between goroutines
// Goroutines communicate by SENDING VALUES THROUGH CHANNELS
// "Don't communicate by sharing memory; share memory by communicating"

// Unbuffered channel — sender BLOCKS until receiver is ready
ch := make(chan string)
go func() { ch <- "hello" }()  // goroutine sends, blocks until received
msg := <-ch                       // main receives

// Buffered channel — sender doesn't block until buffer is full
results := make(chan Hotel, 10)  // buffer of 10 items

// Parallel hotel search — collect results via channel
func searchHotels(cities []string) []Hotel {
    results := make(chan []Hotel, len(cities))

    for _, city := range cities {
        go func(c string) {
            hotels, _ := hotelRepo.FindByCity(c)
            results <- hotels  // send to channel
        }(city)
    }

    // Collect all results
    var allHotels []Hotel
    for range cities {
        hotels := <-results  // receive from channel (blocks until available)
        allHotels = append(allHotels, hotels...)
    }
    return allHotels
}

// Select — choose between multiple channel operations
select {
case result := <-hotelCh:
    fmt.Println("got hotel:", result.Name)
case err := <-errCh:
    fmt.Println("error:", err)
case <-time.After(2 * time.Second):
    fmt.Println("timeout")  // neither channel sent in 2s
}
// Worker pool pattern — limit concurrency
// Problem: 10,000 goroutines all hitting DB = connection exhaustion
// Solution: worker pool limits to N concurrent workers

func processWithWorkerPool(jobs []Job, numWorkers int) {
    jobCh := make(chan Job, len(jobs))
    resultCh := make(chan Result, len(jobs))

    // Start N workers
    for i := 0; i < numWorkers; i++ {
        go func() {
            for job := range jobCh {  // range over channel = receive until closed
                result, err := processJob(job)
                resultCh <- Result{Value: result, Err: err}
            }
        }()
    }

    // Send all jobs
    for _, job := range jobs {
        jobCh <- job
    }
    close(jobCh)  // signals workers to stop after processing remaining jobs

    // Collect results
    for range jobs {
        result := <-resultCh
        // handle result
    }
}

// Mutex — protect shared state (when channels are awkward)
type SafeCounter struct {
    mu    sync.RWMutex
    count int
}

func (c *SafeCounter) Increment() {
    c.mu.Lock()
    defer c.mu.Unlock()  // always unlock, even on panic
    c.count++
}

func (c *SafeCounter) Value() int {
    c.mu.RLock()   // multiple readers OK simultaneously
    defer c.mu.RUnlock()
    return c.count
}

Context — Cancellation, Timeouts, Values
context.Context propagates through every function call in Go
Intermediate▼ 
// Context — carries cancellation signals, deadlines, and request-scoped values
// Rule: ALWAYS accept ctx as the FIRST parameter of every function

func GetHotel(ctx context.Context, id string) (*Hotel, error) {
    // If context is cancelled (request timeout, client disconnect),
    // DB call is also cancelled — no wasted resources
    return db.QueryRowContext(ctx, "SELECT * FROM hotels WHERE id = $1", id)
}

// Timeout context — cancel after 2 seconds
ctx, cancel := context.WithTimeout(context.Background(), 2*time.Second)
defer cancel() // ALWAYS defer cancel to avoid resource leak

hotel, err := GetHotel(ctx, "123")
if err != nil {
    if errors.Is(err, context.DeadlineExceeded) {
        // timed out
    }
}

// Cancellation context — cancel manually
ctx, cancel := context.WithCancel(context.Background())
go func() {
    time.Sleep(5 * time.Second)
    cancel()  // cancel after 5s — all functions using ctx will abort
}()

// Request-scoped values (user ID, request ID, trace ID)
type contextKey string
const userIDKey contextKey = "userID"

ctx = context.WithValue(ctx, userIDKey, "user-123")

// Retrieve value anywhere in the call chain
userID := ctx.Value(userIDKey).(string)

// Listen for cancellation in your own goroutine
func longProcess(ctx context.Context) error {
    for {
        select {
        case <-ctx.Done():
            return ctx.Err()  // context.Canceled or DeadlineExceeded
        default:
            doWork()
        }
    }
}

// Production Go REST API with chi router
package main

import (
    "context"
    "encoding/json"
    "log/slog"
    "net/http"
    "os"
    "os/signal"
    "syscall"
    "time"
    "github.com/go-chi/chi/v5"
    "github.com/go-chi/chi/v5/middleware"
)

func main() {
    r := chi.NewRouter()

    // Middleware
    r.Use(middleware.RequestID)
    r.Use(middleware.RealIP)
    r.Use(middleware.Logger)
    r.Use(middleware.Recoverer)  // recover from panics
    r.Use(middleware.Timeout(60 * time.Second))

    // Routes
    r.Get("/health", func(w http.ResponseWriter, r *http.Request) {
        respondJSON(w, http.StatusOK, map[string]string{"status": "UP"})
    })

    r.Route("/api/v1/hotels", func(r chi.Router) {
        r.Use(authMiddleware)         // apply to all hotel routes
        r.Get("/", listHotels)
        r.Post("/", createHotel)
        r.Get("/{id}", getHotel)
        r.Put("/{id}", updateHotel)
    })

    // Graceful shutdown
    srv := &http.Server{Addr: ":8080", Handler: r}
    go srv.ListenAndServe()

    quit := make(chan os.Signal, 1)
    signal.Notify(quit, syscall.SIGTERM, syscall.SIGINT)
    <-quit  // block until signal received

    ctx, cancel := context.WithTimeout(context.Background(), 30*time.Second)
    defer cancel()
    srv.Shutdown(ctx)
}

// Handler function
func getHotel(w http.ResponseWriter, r *http.Request) {
    id := chi.URLParam(r, "id")
    hotel, err := hotelService.FindByID(r.Context(), id)  // pass request context!
    if err != nil {
        if errors.Is(err, ErrNotFound) {
            respondError(w, http.StatusNotFound, "hotel not found")
            return
        }
        slog.Error("failed to get hotel", "id", id, "err", err)
        respondError(w, http.StatusInternalServerError, "internal error")
        return
    }
    respondJSON(w, http.StatusOK, hotel)
}

// Helper functions
func respondJSON(w http.ResponseWriter, status int, data interface{}) {
    w.Header().Set("Content-Type", "application/json")
    w.WriteHeader(status)
    json.NewEncoder(w).Encode(data)
}
