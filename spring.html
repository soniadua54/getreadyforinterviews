Spring Boot Auto-configuration & IoC
How Spring creates and wires objects for you
Beginner▼
Concept
Code
Architecture
Inversion of Control (IoC)
Normally you create your own objects: HotelService service = new HotelService(new HotelRepository(dataSource)). This couples your code to concrete implementations and makes testing hard. With IoC, Spring creates and wires objects for you — you just declare dependencies.
Dependency Injection (DI) — Spring's implementation of IoC. Instead of creating dependencies, you declare them as constructor parameters. Spring's ApplicationContext (IoC Container) creates all beans at startup and injects them where needed.

Auto-configuration — @SpringBootApplication triggers class-path scanning. If spring-boot-starter-web is on classpath → configures embedded Tomcat and DispatcherServlet. If spring-boot-starter-data-jpa → configures EntityManager and transaction manager. Zero XML required.

@Component / @Service / @Repository

Mark class as a Spring-managed bean. @Service = business logic. @Repository = data access (adds exception translation). All three are semantically equivalent to Spring DI — they differ in purpose/tooling support.

Constructor injection (preferred)

Inject via constructor parameters. Allows final fields. Makes dependencies explicit. Easier to test — just call constructor with mocks. Avoids circular dependencies at startup.

@RestController          // = @Controller + @ResponseBody
@RequestMapping("/api/v1/hotels")
public class HotelController {
    private final HotelService hotelService;

    public HotelController(HotelService service) { this.hotelService = service; }

    @GetMapping("/{id}")
    public ResponseEntity<HotelDto> getHotel(@PathVariable Long id) {
        return hotelService.findById(id)
            .map(ResponseEntity::ok)
            .orElse(ResponseEntity.notFound().build());
    }

    @GetMapping
    public Page<HotelDto> searchHotels(
            @RequestParam String city,
            @RequestParam(required=false, defaultValue="0") Integer minStars,
            @RequestParam(defaultValue="0") int page,
            @RequestParam(defaultValue="20") int size) {
        return hotelService.search(city, minStars, PageRequest.of(page, size));
    }

    @PostMapping
    @ResponseStatus(HttpStatus.CREATED)
    public HotelDto createHotel(@RequestBody @Valid CreateHotelRequest req) {
        return hotelService.create(req);
    }

    @PutMapping("/{id}")
    public HotelDto updateHotel(@PathVariable Long id,
                                @RequestBody @Valid UpdateHotelRequest req) {
        return hotelService.update(id, req);
    }

    @DeleteMapping("/{id}")
    @ResponseStatus(HttpStatus.NO_CONTENT)
    public void deleteHotel(@PathVariable Long id) {
        hotelService.delete(id);
    }
}

@Component
public class BookingDateValidator implements Validator {

        if (req.getCheckOut().isBefore(req.getCheckIn()))

// Global exception handler — handles exceptions across ALL controllers
@RestControllerAdvice
public class GlobalExceptionHandler {

    @ExceptionHandler(BookingNotFoundException.class)
    @ResponseStatus(HttpStatus.NOT_FOUND)
    public ErrorResponse handleNotFound(BookingNotFoundException ex) {
        return new ErrorResponse("NOT_FOUND", ex.getMessage());
    }

    @ExceptionHandler(MethodArgumentNotValidException.class)
    @ResponseStatus(HttpStatus.BAD_REQUEST)
    public ErrorResponse handleValidation(MethodArgumentNotValidException ex) {
        List<String> errors = ex.getBindingResult().getFieldErrors().stream()
            .map(e -> e.getField() + ": " + e.getDefaultMessage())
            .collect(toList());
        return new ErrorResponse("VALIDATION_ERROR", "Validation failed", errors);
    }

    @ExceptionHandler(Exception.class)
    @ResponseStatus(HttpStatus.INTERNAL_SERVER_ERROR)
    public ErrorResponse handleGeneral(Exception ex) {
        log.error("Unhandled exception", ex);
        return new ErrorResponse("INTERNAL_ERROR", "An unexpected error occurred");
    }
}

@Entity
@Table(name = "bookings")
public class Booking {
    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Long id;

    @Column(nullable = false)
    private Long userId;

    @Column(nullable = false)
    private String roomId;

    @Enumerated(EnumType.STRING) // store as "CONFIRMED" not 0
    private BookingStatus status;

    @Column(precision = 10, scale = 2)
    private BigDecimal totalPrice;

    @CreatedDate                // auto-set by Spring Data
    private LocalDateTime createdAt;

    @LastModifiedDate
    private LocalDateTime updatedAt;

    @Version                    // optimistic locking — auto-increments on update
    private Long version;

    // Relationship — lazy load rooms (don't eager load by default!)
    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "hotel_id")
    private Hotel hotel;

    @OneToMany(mappedBy = "booking", cascade = CascadeType.ALL, orphanRemoval = true)
    private List<Payment> payments = new ArrayList<>();
}
// Spring Data generates queries from method names
public interface BookingRepository extends JpaRepository<Booking, Long> {

    // Generated: SELECT * FROM bookings WHERE user_id = ? ORDER BY created_at DESC
    List<Booking> findByUserIdOrderByCreatedAtDesc(Long userId);

    // Pagination built-in
    Page<Booking> findByStatus(BookingStatus status, Pageable pageable);

    // Custom JPQL query
    @Query("SELECT b FROM Booking b WHERE b.hotel.city = :city AND b.checkIn >= :date")
    List<Booking> findByCityAndDate(@Param("city") String city, @Param("date") LocalDate date);

    // JOIN FETCH — solves N+1 by loading association in one query
    @Query("SELECT b FROM Booking b JOIN FETCH b.hotel WHERE b.userId = :userId")
    List<Booking> findWithHotelByUserId(@Param("userId") Long userId);

    // Native SQL for complex queries
    @Query(value = "SELECT * FROM bookings WHERE check_in BETWEEN ?1 AND ?2",
           nativeQuery = true)
    List<Booking> findByCheckInBetween(LocalDate from, LocalDate to);

    // Modifying query (UPDATE/DELETE)
    @Modifying
    @Query("UPDATE Booking b SET b.status = :status WHERE b.id = :id")
    int updateStatus(@Param("id") Long id, @Param("status") BookingStatus status);

    // Exists check (more efficient than findById() != null)
    boolean existsByUserIdAndRoomId(Long userId, String roomId);

    // Count
    long countByStatus(BookingStatus status);
}
@Service
public class BookingService {

    // Default: REQUIRED — joins existing tx or creates new
    @Transactional
    public Booking createBooking(CreateBookingRequest req) {
        Booking booking = new Booking(req);
        booking = bookingRepo.save(booking);
        paymentService.charge(booking); // same transaction
        return booking;
    }

    // REQUIRES_NEW — new transaction regardless of outer tx
    // Use for: audit logging (must commit even if outer tx rolls back)
    @Transactional(propagation = Propagation.REQUIRES_NEW)
    public void logAuditEvent(AuditEvent event) {
        auditRepo.save(event);
    }

    // readOnly — optimizes for reads (no dirty checking, can use read replicas)
    @Transactional(readOnly = true)
    public Page<BookingDto> findBookings(Long userId, Pageable p) {
        return bookingRepo.findByUserId(userId, p).map(bookingMapper::toDto);
    }

    // rollbackFor — explicitly roll back on checked exceptions too
    @Transactional(rollbackFor = Exception.class)
    public void processRefund(Long bookingId) throws RefundException {
        // if RefundException (checked) is thrown, rolls back
    }
}

// COMMON TRAP: calling @Transactional method from same class bypasses proxy!
// Internal calls don't go through Spring's AOP proxy.
// Fix: inject self, or use ApplicationContext.getBean(), or restructure
@Service
public class BookingService {
    @Autowired private BookingService self; // inject self — ugly but works
    public void outer() { self.inner(); } // goes through proxy
    @Transactional public void inner() { }
}

// Enable caching in your application
@SpringBootApplication
@EnableCaching
public class Application { public static void main(String[] a) { SpringApplication.run(Application.class, a); } }

@Service
public class HotelService {

    // @Cacheable: cache result, skip method on cache hit
    @Cacheable(value = "hotels", key = "#id",
               unless = "#result == null")  // don't cache nulls
    public Hotel findById(Long id) {
        return hotelRepo.findById(id).orElse(null);
    }

    // @CachePut: always run method AND update cache (for updates)
    @CachePut(value = "hotels", key = "#result.id")
    public Hotel updateHotel(Long id, UpdateHotelRequest req) {
        Hotel hotel = hotelRepo.findById(id).orElseThrow();
        hotel.update(req);
        return hotelRepo.save(hotel);
    }

    // @CacheEvict: remove from cache on delete
    @CacheEvict(value = "hotels", key = "#id")
    public void deleteHotel(Long id) { hotelRepo.deleteById(id); }

    // Evict all entries in a cache (e.g., after bulk update)
    @CacheEvict(value = "hotels", allEntries = true)
    public void refreshAllHotels() { /* trigger refresh */ }

    // Complex key expression with SpEL
    @Cacheable(value = "search", key = "#city + ':' + #checkIn + ':' + #checkOut")
    public List<Hotel> search(String city, LocalDate checkIn, LocalDate checkOut) {
        return hotelRepo.search(city, checkIn, checkOut);
    }
}

// Programmatic cache access (when annotation isn't flexible enough)
@Service
public class AvailabilityService {
    private final RedisTemplate<String, Integer> redisTemplate;

    public Optional<Integer> getCachedAvailability(String roomId, LocalDate date) {
        String key = "avail:" + roomId + ":" + date;
        Integer count = redisTemplate.opsForValue().get(key);
        return Optional.ofNullable(count);
    }

    public void setAvailability(String roomId, LocalDate date, int count) {
        String key = "avail:" + roomId + ":" + date;
        redisTemplate.opsForValue().set(key, count, Duration.ofMinutes(1));
    }
}

Spring Security Filter Chain with JWT
HTTP Request ↓ [JwtAuthenticationFilter] ← extracts + validates JWT from Authorization header ↓ [SecurityContextHolder] ← stores authenticated user for this request ↓ [AuthorizationFilter] ← checks @PreAuthorize / hasRole() / hasAuthority() ↓ [Controller method] ← only reached if authorized JWT Token Structure: Header.Payload.Signature {alg:HS256}.{userId,role,exp}.{HMAC(header+payload, secret)} Claims you typically store: sub: userId email: user@example.com roles: ["ROLE_USER", "ROLE_ADMIN"] exp: 1719910800 (expiry timestamp)
@Configuration
@EnableWebSecurity
@EnableMethodSecurity
public class SecurityConfig {

    @Bean
    public SecurityFilterChain filterChain(HttpSecurity http) throws Exception {
        return http
            .csrf(csrf -> csrf.disable())       // stateless API — no CSRF
            .sessionManagement(s -> s.sessionCreationPolicy(STATELESS))
            .authorizeHttpRequests(auth -> auth
                .requestMatchers("/api/v1/auth/**").permitAll()
                .requestMatchers("/api/v1/hotels/**").permitAll()        // public
                .requestMatchers("/api/v1/bookings/**").authenticated()
                .requestMatchers("/api/v1/admin/**").hasRole("ADMIN")
                .anyRequest().authenticated()
            )
            .addFilterBefore(jwtFilter, UsernamePasswordAuthenticationFilter.class)
            .build();
    }
}

// JWT filter — validates token on every request
@Component
public class JwtAuthenticationFilter extends OncePerRequestFilter {
    private final JwtService jwtService;

    @Override
    protected void doFilterInternal(HttpServletRequest req, HttpServletResponse res,
                                      FilterChain chain) throws Exception {
        String header = req.getHeader("Authorization");
        if (header == null || !header.startsWith("Bearer ")) {
            chain.doFilter(req, res); return;
        }
        String token = header.substring(7);
        if (jwtService.isValid(token)) {
            String userId = jwtService.extractUserId(token);
            List<GrantedAuthority> authorities = jwtService.extractRoles(token);
            UsernamePasswordAuthenticationToken auth =
                new UsernamePasswordAuthenticationToken(userId, null, authorities);
            SecurityContextHolder.getContext().setAuthentication(auth);
        }
        chain.doFilter(req, res);
    }
}

Kafka event flow in booking system
Booking Service Notification Service │ │ @Transactional createBooking() │ │ │ 1. Save booking to DB │ 2. KafkaTemplate.send("booking.events") │ │ ──────────────────────────────────► │ │ @KafkaListener │ 3. Process event │ 4. Send email/SMS/push │ 5. Commit offset Topics: booking.events → partitioned by userId (ordered per user) payment.events → partitioned by bookingId hotel.updates → partitioned by hotelId (for search index sync) Dead Letter Topic: booking.events.DLT ← failed after 3 retries, alert on-call
// Producer
@Service
public class BookingEventProducer {
    private final KafkaTemplate<String, BookingEvent> kafkaTemplate;

    public void publishBookingCreated(Booking booking) {
        var event = new BookingEvent(booking.getId(), "CREATED", Instant.now());
        var record = new ProducerRecord<>(
            "booking.events",
            booking.getUserId().toString(), // partition key → same user in order
            event
        );
        kafkaTemplate.send(record)
            .whenComplete((result, ex) -> {
                if (ex != null) log.error("Failed to publish event", ex);
                else log.info("Event sent: offset={}", result.getRecordMetadata().offset());
            });
    }
}

// Consumer with error handling and retry
@Component
public class BookingEventConsumer {

    @KafkaListener(topics = "booking.events", groupId = "notification-service",
                   containerFactory = "kafkaListenerContainerFactory")
    public void handleBookingEvent(BookingEvent event,
                                    @Header(KafkaHeaders.RECEIVED_PARTITION) int partition,
                                    @Header(KafkaHeaders.OFFSET) long offset) {
        log.info("Processing event bookingId={} partition={} offset={}",
                 event.getBookingId(), partition, offset);
        notificationService.sendBookingConfirmation(event);
        // offset committed automatically after method returns without exception
    }
}

// Error handling config — retry 3 times, then DLT
@Bean
public DefaultErrorHandler errorHandler(KafkaTemplate<?, ?> kafkaTemplate) {
    DeadLetterPublishingRecoverer recoverer = new DeadLetterPublishingRecoverer(kafkaTemplate);
    ExponentialBackOff backOff = new ExponentialBackOff(1000L, 2.0);
    backOff.setMaxAttempts(3);
    return new DefaultErrorHandler(recoverer, backOff);
    // Retry: 1s → 2s → 4s → DLT (booking.events.DLT)
}

// Add to build.gradle: spring-cloud-starter-circuitbreaker-resilience4j

@Service
public class HotelService {

    // Circuit breaker — stops calling payment if it's failing
    @CircuitBreaker(name = "payment", fallbackMethod = "paymentFallback")
    public PaymentResult processPayment(PaymentRequest req) {
        return paymentClient.charge(req);
    }

    public PaymentResult paymentFallback(PaymentRequest req, Exception ex) {
        log.warn("Payment service unavailable, using fallback", ex);
        return PaymentResult.pending("Will retry when service recovers");
    }

    // Retry — retry on transient failures (not on 4xx)
    @Retry(name = "hotelSearch", fallbackMethod = "searchFallback")
    public List<Hotel> searchHotels(SearchRequest req) {
        return elasticsearchClient.search(req);
    }

    // Rate limiter — limit to 50 requests/second
    @RateLimiter(name = "hotelApi")
    public Hotel getHotel(String id) { return hotelRepo.findById(id); }

    // Bulkhead — limit concurrent calls to avoid thread exhaustion
    @Bulkhead(name = "payment", type = Bulkhead.Type.THREADPOOL)
    public CompletableFuture<PaymentResult> chargeAsync(PaymentRequest req) {
        return CompletableFuture.supplyAsync(() -> paymentClient.charge(req));
    }
}

@Service
public class BookingService {
    private final Counter bookingsCreated;
    private final Timer bookingDuration;
    private final DistributionSummary bookingValue;

    public BookingService(MeterRegistry registry) {
        bookingsCreated = Counter.builder("bookings.created")
            .description("Total bookings created")
            .tag("service", "hotel-booking")
            .register(registry);

        bookingDuration = Timer.builder("booking.creation.duration")
            .description("Time to create a booking")
            .percentiles(0.5, 0.95, 0.99) // p50, p95, p99 latency
            .register(registry);

        bookingValue = DistributionSummary.builder("booking.value")
            .description("Booking price distribution")
            .baseUnit("USD")
            .register(registry);
    }

    public Booking createBooking(CreateBookingRequest req) {
        return bookingDuration.record(() -> {
            Booking booking = doCreateBooking(req);
            bookingsCreated.increment();
            bookingValue.record(booking.getTotalPrice().doubleValue());
            return booking;
        });
    }
}
