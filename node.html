Event Loop — How Node.js Actually Works
The most important concept — everything else follows from this
Beginner▼
How it works
Code examples
Loop phases
Single-threaded but non-blocking
Node.js runs JavaScript on a single thread. It cannot do two things at the same time. But it can initiate a slow operation (file read, DB query, HTTP call) and immediately move on to other work. When the slow operation completes, a callback is queued and executed when the thread is free.
Event loop — simplified
JavaScript Call Stack (single thread) ↓ [Current executing code] ↓ (when empty) [Event Queue / Callback Queue] ↓ picks next callback [Execute callback] → might add more async ops Libuv Thread Pool (default 4 threads): handles: file I/O, DNS lookup, crypto, compression When done: callback goes into Event Queue Network I/O: handled by OS kernel directly (epoll/kqueue) Database queries, HTTP requests → no thread pool needed
The key insight: Node.js is NOT fast because of single thread. It's fast because it never WAITS. While your DB query runs (in the OS kernel), Node's thread is free to handle 10,000 other requests. This is why Node excels at I/O-heavy workloads like API servers — which is exactly what Secure Media Service was.
When Node.js is wrong: CPU-intensive work (image resizing, video encoding, heavy computation) BLOCKS the single thread. Everything else stops. For CPU work: use Worker Threads or delegate to Go/Java service (like your Walmart architecture did).
// Demonstrating non-blocking I/O
const fs = require('fs');

// This does NOT block the thread
fs.readFile('/large/file.txt', (err, data) => {
    // This runs LATER, when OS has finished reading
    console.log('File read complete:', data.length);
});

// This runs IMMEDIATELY while file is being read
console.log('This runs first');
// Output: "This runs first" then "File read complete: ..."

// Promise-based (modern way)
const fsPromises = require('fs/promises');

async function readFile() {
    const data = await fsPromises.readFile('/large/file.txt');
    console.log('Done:', data.length);
}

Event loop phases (in order)
Phase 1: timers Runs expired setTimeout() and setInterval() callbacks Phase 2: pending callbacks I/O callbacks deferred from previous iteration Phase 3: idle, prepare (internal) Phase 4: poll Retrieve new I/O events and execute callbacks Wait here if no timers pending Phase 5: check setImmediate() callbacks execute here Phase 6: close callbacks socket.on('close', ...) etc Between each phase: process.nextTick() callbacks (highest priority) Promise microtasks (.then, .catch, async/await)
process.nextTick vs setImmediate: process.nextTick() runs before the NEXT event loop phase (higher priority). setImmediate() runs in the check phase. Use process.nextTick() for deferring within same phase. Use setImmediate() to let I/O callbacks run first.

// Complete Express app structure (production-grade)
// src/app.js — application factory
const express = require('express');
const helmet = require('helmet');         // security headers
const cors = require('cors');             // CORS
const rateLimit = require('express-rate-limit');
const { json } = require('express');

function createApp() {
    const app = express();

    // Security middleware
    app.use(helmet());
    app.use(cors({ origin: process.env.ALLOWED_ORIGINS?.split(',') }));

    // Rate limiting
    app.use('/api', rateLimit({
        windowMs: 15 * 60 * 1000,   // 15 minutes
        max: 100,                      // 100 requests per window
        standardHeaders: true,
        message: { error: 'Too many requests' }
    }));

    // Body parsing
    app.use(json({ limit: '10mb' }));
    app.use(express.urlencoded({ extended: true }));

    // Routes
    app.use('/api/v1/hotels', require('./routes/hotels'));
    app.use('/api/v1/bookings', require('./routes/bookings'));
    app.get('/health', (req, res) => res.json({ status: 'UP' }));

    // Global error handler (MUST have 4 params)
    app.use(require('./middleware/errorHandler'));

    return app;
}

// Middleware chain — each function calls next() to pass control
// Request → middleware1 → middleware2 → route handler → response

// Authentication middleware
const jwt = require('jsonwebtoken');

function authenticate(req, res, next) {
    const header = req.headers['authorization'];
    if (!header?.startsWith('Bearer ')) {
        return res.status(401).json({ error: 'No token provided' });
    }
    try {
        const token = header.slice(7);
        req.user = jwt.verify(token, process.env.JWT_SECRET);
        next(); // pass to next middleware
    } catch (err) {
        res.status(401).json({ error: 'Invalid token' });
    }
}

// Validation middleware with Zod
const { z } = require('zod');

const createBookingSchema = z.object({
    roomId: z.string().min(1),
    checkIn: z.string().regex(/^\d{4}-\d{2}-\d{2}$/),
    checkOut: z.string().regex(/^\d{4}-\d{2}-\d{2}$/),
    guestCount: z.number().min(1).max(10),
}).refine(data => data.checkOut > data.checkIn, {
    message: 'checkOut must be after checkIn',
    path: ['checkOut']
});

function validate(schema) {
    return (req, res, next) => {
        const result = schema.safeParse(req.body);
        if (!result.success) {
            return res.status(400).json({ error: result.error.flatten() });
        }
        req.body = result.data; // replace with parsed/typed data
        next();
    };
}

// Request logging middleware
function requestLogger(req, res, next) {
    const start = Date.now();
    res.on('finish', () => {
        console.log(`${req.method} ${req.path} ${res.statusCode} ${Date.now()-start}ms`);
    });
    next();
}
// Global error handler — MUST have exactly 4 parameters
// Express identifies it as error handler by arity (4 params)
function errorHandler(err, req, res, next) {
    // Log with context
    console.error({
        message: err.message,
        stack: err.stack,
        path: req.path,
        method: req.method,
        requestId: req.id,
    });

    // Known operational errors
    if (err.name === 'ValidationError') {
        return res.status(400).json({ error: err.message, details: err.details });
    }
    if (err.name === 'NotFoundError') {
        return res.status(404).json({ error: err.message });
    }
    if (err.name === 'UnauthorizedError') {
        return res.status(401).json({ error: 'Unauthorized' });
    }

    // Unknown errors — don't leak stack trace to client
    res.status(500).json({ error: 'Internal server error' });
}

// Custom error classes
class AppError extends Error {
    constructor(message, statusCode = 500) {
        super(message);
        this.statusCode = statusCode;
        this.isOperational = true; // expected error, not a bug
    }
}
class NotFoundError extends AppError {
    constructor(msg) { super(msg, 404); this.name = 'NotFoundError'; }
}

// Async wrapper to avoid try/catch in every route handler
const asyncHandler = fn => (req, res, next) =>
    Promise.resolve(fn(req, res, next)).catch(next);

// Usage:
router.get('/:id', asyncHandler(async (req, res) => {
    const hotel = await hotelService.findById(req.params.id);
    if (!hotel) throw new NotFoundError(`Hotel ${req.params.id} not found`);
    res.json(hotel);
}));

AspectRESTGraphQL
EndpointsOne per resource (/hotels, /rooms)Single endpoint (/graphql)

Over-fetchingCommon (get all fields, need 3)Never (only get what you ask for)

CachingEasy (HTTP cache by URL)Hard (all POSTs, need custom)
Learning curveLowMedium
Why Walmart used GraphQL for VTO: The Virtual Try-On feature needed hotel/product data + user preferences + model data in one request. With REST you'd need 3+ requests. GraphQL let the frontend specify exactly what combination it needed — critical for mobile where each extra request costs latency.
// GraphQL schema (SDL — Schema Definition Language)
const typeDefs = `
  type Hotel {
    id: ID!
    name: String!
    city: String!
    starRating: Int!
    rooms: [Room!]!           # nested — fetched via resolver
    reviewScore: Float
  }

  type Room {
    id: ID!
    type: String!
    price: Float!
    availableCount: Int!
  }

  type Query {
    hotel(id: ID!): Hotel
    hotels(city: String!, minStars: Int): [Hotel!]!
    searchHotels(input: SearchInput!): SearchResult!
  }

  input SearchInput {
    city: String!
    checkIn: String!
    checkOut: String!
    guestCount: Int!
  }

  type SearchResult {
    hotels: [Hotel!]!
    totalCount: Int!
  }

  type Mutation {
    createBooking(input: CreateBookingInput!): Booking!
    cancelBooking(id: ID!): Boolean!
  }

  input CreateBookingInput {
    roomId: ID!
    checkIn: String!
    checkOut: String!
    guestCount: Int!
  }
`;

// Resolvers — functions that fetch/compute each field
const resolvers = {
    Query: {
        hotel: async (_, { id }, context) => {
            return context.dataSources.hotelAPI.getById(id);
        },
        hotels: async (_, { city, minStars }, context) => {
            return context.dataSources.hotelAPI.search({ city, minStars });
        },
    },
    Hotel: {
        // Resolver for the rooms field — only called if client asks for rooms
        rooms: async (hotel, _, context) => {
            return context.dataSources.roomAPI.getByHotelId(hotel.id);
        },
    },
    Mutation: {
        createBooking: async (_, { input }, context) => {
            requireAuth(context); // check authentication
            return context.dataSources.bookingAPI.create(input, context.user.id);
        },
    },
};
// N+1 problem in GraphQL
// Query: { hotels { id name rooms { price } } }
// Without DataLoader: 1 query for hotels + N queries for each hotel's rooms

// Solution: DataLoader — batches and caches all room requests into ONE query
const DataLoader = require('dataloader');

// Create loader — batch function receives array of ALL requested hotel IDs
const roomLoader = new DataLoader(async (hotelIds) => {
    // ONE database call for all hotel IDs
    const rooms = await db.query(
        'SELECT * FROM rooms WHERE hotel_id = ANY($1)',
        [hotelIds]
    );
    // Return arrays in SAME ORDER as input hotelIds
    return hotelIds.map(id => rooms.filter(r => r.hotel_id === id));
});

// In resolver — DataLoader batches all calls within same tick
Hotel: {
    rooms: (hotel, _, context) => {
        return context.roomLoader.load(hotel.id); // batched automatically
    }
}

// Result: 10 hotels → 1 query for hotels + 1 batch query for ALL rooms
// Without DataLoader: 1 + 10 = 11 queries
// With DataLoader: 1 + 1 = 2 queries

Streams — Processing Large Data
Read, Transform, Write streams without loading everything into memory
Intermediate▼ 
// Streams process data chunk by chunk — never load entire file into memory
// Critical for: file upload/download, CSV processing, large data pipelines

// Problem without streams: reading a 1GB file
const data = fs.readFileSync('1gb-file.csv'); // 1GB in RAM — crash!

// Solution with streams
const { Transform } = require('stream');
const { pipeline } = require('stream/promises');
const csv = require('csv-parse');

// Process 1GB CSV file with minimal memory
async function processLargeCSV(filePath) {
    const readStream = fs.createReadStream(filePath);    // read chunks

    const parseCSV = csv({ columns: true, skip_empty_lines: true });

    const transform = new Transform({
        objectMode: true,
        transform(row, encoding, callback) {
            // Process one row at a time — constant memory usage
            const processed = {
                id: row.id,
                price: parseFloat(row.price),
                city: row.city.toLowerCase(),
            };
            this.push(processed); // pass to next step
            callback();
        }
    });

    const results = [];
    const collect = new Transform({
        objectMode: true,
        transform(row, _, cb) { results.push(row); cb(); }
    });

    // pipeline: automatically handles backpressure and errors
    await pipeline(readStream, parseCSV, transform, collect);
    return results;
}

// HTTP streaming response (large file download)
app.get('/download/:id', async (req, res) => {
    res.setHeader('Content-Type', 'application/octet-stream');
    res.setHeader('Content-Disposition', `attachment; filename="${req.params.id}"`);

    // Stream from S3 directly to response — no memory buffer
    const s3Stream = s3.getObject({ Bucket: 'media', Key: req.params.id }).createReadStream();
    s3Stream.pipe(res);

    s3Stream.on('error', err => {
        console.error('Stream error:', err);
        res.status(500).end();

Clustering & Worker Threads
Scale Node.js across CPU cores
Advanced▼ 
// Problem: Node.js uses 1 CPU core by default
// Your server has 8 cores — 7 are idle

// Solution 1: Cluster module — multiple processes sharing port
const cluster = require('cluster');
const os = require('os');

if (cluster.isPrimary) {
    const cpuCount = os.cpus().length;
    console.log(`Primary ${process.pid}: forking ${cpuCount} workers`);

    for (let i = 0; i < cpuCount; i++) {
        cluster.fork();
    }

    cluster.on('exit', (worker, code, signal) => {
        console.log(`Worker ${worker.pid} died. Restarting...`);
        cluster.fork(); // auto-restart crashed workers
    });
} else {
    // Each worker runs the app independently
    const app = require('./app');
    app.listen(3000, () => {
        console.log(`Worker ${process.pid} listening on port 3000`);
    });
}

// Solution 2: Worker Threads — CPU-intensive tasks off main thread
const { Worker, isMainThread, parentPort, workerData } = require('worker_threads');

if (isMainThread) {
    // Main thread — handles requests, delegates CPU work
    function resizeImageAsync(imagePath, options) {
        return new Promise((resolve, reject) => {
            const worker = new Worker('./imageWorker.js', {
                workerData: { imagePath, options }
            });
            worker.on('message', resolve);
            worker.on('error', reject);
        });
    }
    // Main thread stays responsive while worker resizes image
} else {
    // Worker thread — does the CPU work
    const sharp = require('sharp');
    sharp(workerData.imagePath)
        .resize(workerData.options)
        .toBuffer()
        .then(buffer => parentPort.postMessage(buffer));
}
